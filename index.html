Iris Modal Library

Powerful and flexible library for creating Bootstrap 5 modal dialogs with advanced features

v1.0.6 2025-02-15 Igor Lovri캖 MIT License
Fast & Easy

Simple API for creating dialogs in one line of code

Customizable

Different sizes, types and styles of dialogs

Multi-language

Support for multiple languages with easily extensible translations

Basic Examples

Alert Dialog

Simple informative message with OK button.


Iris.alert('This is an informative message!');
Confirm Dialog

Confirmation with Yes/No buttons that returns a Promise.


Iris.confirm('Are you sure you want to continue?')
    .then(result => {
        if (result) Iris.success('You confirmed!');
        else Iris.alert('You cancelled.');
    }
);
Success Dialog

Display a success message.


Iris.success('Operation completed successfully!');
Error Dialog

Display an error with red style.


Iris.error('An error occurred during processing!');
Prompt Dialog

Text input from user with Promise result.


Iris.prompt('Enter your name:', 'Name Input', 'User').then(value => {
    if (value !== null) Iris.success('Hello, ' + value + '!');
});
Warning Dialog

Warning with yellow style.


Iris.warning('Attention! This action cannot be undone.');

Dialog Sizes

Different Modal Sizes

Iris supports different dialog sizes using constant values.


// Small size
Iris.show({ title: 'Small dialog', size: Iris.SIZE_SMALL });

// Normal size (default)
Iris.show({ title: 'Normal dialog', size: Iris.SIZE_NORMAL });

// Large size
Iris.show({ title: 'Large dialog', size: Iris.SIZE_LARGE });

// Extra large size
Iris.show({ title: 'XL dialog', size: Iris.SIZE_XLARGE });

// Full width
Iris.show({ title: 'Full width', size: Iris.SIZE_FULLWIDTH });

// Fullscreen
Iris.show({ title: 'Fullscreen', size: Iris.SIZE_FULLSCREEN });

Header Styles

Different Types (Header Colors)

Style the dialog header with Bootstrap colors.


Iris.show({
    title: 'Primary style',
    type: Iris.TYPE_PRIMARY,
    message: 'Blue header for primary actions.'
});

Advanced Features

Custom Buttons

Create dialogs with custom buttons and actions.


Iris.show({
    title: 'Choose Action',
    message: '
What would you like to do with this document?
',
    type: Iris.TYPE_INFO,
    buttons: [
        {
            id: "btn_save",
            label: 'Save...',
            cssClass: 'btn-success',
            icon: 'bi bi-save',
            autoSpin: true,
            autoClose: false,
            action: (dialogRef) => {
                dialogRef.enableButtons(false);

                setTimeout(() => {
                    dialogRef.buttonSpin('btn_save', false);
                    dialogRef.enableButtons(false);
                    Iris.success('Document saved!')
                    dialogRef.close();
                }, 1000);
            }
        },
        {
            label: 'Print',
            cssClass: 'btn-primary',
            icon: 'bi bi-printer',
            action: () => Iris.alert('Sending to printer...')
        },
        {
            label: 'Delete',
            cssClass: 'btn-danger',
            icon: 'bi bi-trash',
            action: () => Iris.warning('Document deleted!')
        },
        {
            label: 'Cancel',
            cssClass: 'btn-secondary'
        }
    ]
});
Multiple modals

Open multiple dialogs.


function showModal (x) {
    var sizes = [Iris.SIZE_LARGE, Iris.SIZE_NORMAL, Iris.SIZE_SMALL];
    Iris.show({
        title: `Dialog #${x}`,
        message: `
This is dialog #${x}
`,
        draggable: true,
        size: sizes[x-1],
        buttons: [{
            label: 'Ok',
            cssClass: 'btn-primary',
            action: function (dialogRef) {
                dialogRef.close();
            }
        }]
    });
}

var x=1;
execShowModal(x);
var timer=setInterval(function () {
    x++;
    execShowModal(x);
    if (x==3) { clearInterval(timer); }
}, 1000);

Callback Functions

React to open, close and other dialog events.


Iris.show({
    title: 'Dialog with Callbacks',
    message: '
Open Developer Console (F12) to see messages!
',
    type: Iris.TYPE_INFO,
    onshow: () => console.log('游댯 onshow: Dialog will be shown'),
    onshown: () => console.log('游릭 onshown: Dialog is shown'),
    onhide: () => { console.log('游리 onhide: Dialog will be closed'); return true; },
    onhidden: () => console.log('游댮 onhidden: Dialog is closed')
});

Dynamic Content

Update Content

Dynamically change dialog content, title and buttons.


Iris.show({
    title: 'Original Title',
    message: 'Original content',
    buttons: [
        {
            label: 'Update',
            cssClass: 'btn-primary',
            id: 'btnUpdate',
            action: (d) => {
                d.setContent('
Content updated!
');
                d.setTitle('New Title');
                d.disableButton('btnUpdate');
                return false;
            }
        },
        { label: 'Close', cssClass: 'btn-secondary' }
    ]
});
Progress Bar

Display progress inside the dialog.


let dialog=Iris.show({
    title: 'Loading File...',
    message: `
0%
Please wait...
`,
    type: Iris.TYPE_INFO,
    closeButton: false,
    keyboard: false,
    backdrop: 'static'
});

let progress = 0;
const interval = setInterval(() => {
    progress += 10;
    const bar = document.getElementById('progressBar');
    if (bar) {
        bar.style.width = progress + '%';
        bar.textContent = progress + '%';
    }
    if (progress >= 100) {
        clearInterval(interval);
        setTimeout(() => {
            dialog.close();
            Iris.success('File loaded successfully!');
        }, 500);
    }
}, 400);

Multi-language Support

Language Switching

Iris supports multi-language. You can easily add new languages and switch between them.


if (!Iris.i18n['de-DE']) {
    Iris.i18n['de-DE'] = {
        name: 'Deutsch',
        translations: { ok: 'OK', cancel: 'Abbrechen', yes: 'Ja', no: 'Nein', close: 'Schlie른n', confirm: 'Best칛tigen', info: 'Information', success: 'Erfolg', warning: 'Warnung', error: 'Fehler' }
    };
}
Iris.setLanguage('de-DE');
Iris.confirm('M칬chten Sie fortfahren?').then(result => {
    if (result) Iris.success('Sie haben best칛tigt!');
    Iris.setLanguage('en-US');
});

Installation and Usage

How to Use Iris?
1. Include Bootstrap 5

<link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/css/bootstrap.min.css" rel="stylesheet">
<script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/js/bootstrap.bundle.min.js"></script>
2. Include Iris Files

<link rel="stylesheet" href="iris.css">
<script src="iris.js"></script>
3. Use Iris

// Simple
Iris.alert('Hello World!');

// Advanced
const dialog = new Iris();
dialog.show({
    title: 'My Dialog',
    message: 'Dialog content',
    size: Iris.SIZE_LARGE,
    type: Iris.TYPE_PRIMARY,
    buttons: [{ label: 'OK', cssClass: 'btn-primary' }]
});
